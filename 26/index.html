<script>

const target = new Date("January 31, 2026 00:00:00").getTime();

const d=document.getElementById('days'),h=document.getElementById('hours'),m=document.getElementById('minutes'),s=document.getElementById('seconds');

const spotifyFrame = document.getElementById('spotifyFrame');

const midnightMessage = document.getElementById('midnightMessage');

function setSpotifySource() {

  const now = new Date();

  const isBirthday = now.getFullYear() === 2026 && now.getMonth() === 0 && now.getDate() === 31;

  if (isBirthday) {

    spotifyFrame.src = "https://open.spotify.com/embed/track/0nlERwr8m4Ks1RYYuTVm0C";

  } else {

    spotifyFrame.src = "https://open.spotify.com/embed/playlist/1tFYNhuVFiMOPSlUZikOxu";

  }

}

function tick(){

  const now=Date.now(),dist=target-now;

  if(dist <= 0){

    midnightMessage.style.display = 'flex';

  }

  d.textContent=Math.max(0, Math.floor(dist/86400000));

  h.textContent=Math.max(0, Math.floor(dist/3600000)%24);

  m.textContent=Math.max(0, Math.floor(dist/60000)%60);

  s.textContent=Math.max(0, Math.floor(dist/1000)%60);

}

setSpotifySource();

tick(); setInterval(tick,1000);

// Confetti

function confettiLoop(){

  for(let i=0;i<120;i++){

    const c=document.createElement('div');

    c.className='confetti';

    c.style.left=Math.random()*100+'vw';

    c.style.animationDuration=3+Math.random()*3+'s';

    c.style.background=['#ff5fa2','#ff8acb','#ffd1e8','#fff'][Math.floor(Math.random()*4)];

    document.body.appendChild(c);

    setTimeout(()=>c.remove(),6000);

  }

}

confettiLoop(); setInterval(confettiLoop,10000);

// Lightbox

const lightbox=document.getElementById('lightbox'),lightImg=document.getElementById('lightboxImg');

document.querySelectorAll('.gallery img').forEach(img=>{

  img.onclick=()=>{ lightImg.src=img.src; lightbox.style.display='flex'; };

});

lightbox.onclick=()=>lightbox.style.display='none';

// Spotify Button

const btn=document.getElementById('musicBtn');

const player=document.getElementById('spotifyPlayer');

let opened = false;

document.addEventListener('click', () => {

  if(!opened){ player.style.display='block'; opened=true; }

},{ once:true });

btn.onclick = ()=>{ player.style.display = player.style.display==='none' ? 'block' : 'none'; };
